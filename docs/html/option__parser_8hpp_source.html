325  std::vector<std::vector<std::string>> parsed_vals;
326 
327  // Remove outer brackets
328  unsigned firstBracket = s.find("[");
329  unsigned lastBracket = s.find_last_of("]");
330  s.erase(firstBracket, 1);
331  s.erase(s.begin() + lastBracket - 1, s.end());
332 
333  // Process inner brackets
334  while (s.find("[") != std::string::npos) {
335  std::string sCopy = s;
336  firstBracket = sCopy.find("[");
337  lastBracket = sCopy.find("]");
338  sCopy.erase(firstBracket, 1);
339  sCopy.erase(sCopy.begin() + lastBracket - 1, sCopy.end());
340  // Process subsstring
341  std::vector<std::string> elems;
342  std::stringstream ss(sCopy);
343  std::string item;
344  while (getline(ss, item, ',')) {
345  item = std::regex_replace(item, std::regex("\\s+"), "");
346  elems.push_back(item);
347  }
348 
349  parsed_vals.push_back(elems);
350  s.erase(s.begin(), s.begin() + lastBracket + 1);
351  if (s.find(",") == std::string::npos)
352  break;
353  unsigned commaPos = s.find(",");
354  s.erase(s.begin(), s.begin() + commaPos + 1);
