326  virtual bool updateNextBelief_(FilterResultPtr filterResult, abt::BeliefNode *nextBelief) {
327  const ABTExtendedOptions *abtOptions = static_cast<const ABTExtendedOptions *>(problemEnvironmentOptions_);
328  if (!abtOptions->hasChanges && !abtOptions->interactive) {
