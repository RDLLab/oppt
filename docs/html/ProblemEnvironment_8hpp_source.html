400  void updateViewer(const RobotStateSharedPtr & currentState,
401  const VectorRobotStatePtr& particles = VectorRobotStatePtr()) {
402  if (!currentState) {
403  LOGGING("currentState is null. Cannot visualize current simulation state");
404  return;
405  }
406  VectorRobotStatePtr subStates = currentState->getSubStates();
407  unsigned int numDrawnParticles = problemEnvironmentOptions_->particlePlotLimit;
408  if (particles.size() < numDrawnParticles)
409  numDrawnParticles = particles.size();
410  for (size_t i = 0; i != subStates.size(); ++i) {
411  FloatType d = ((1.0 / (FloatType)(subStates.size())) * i);
412  VectorRobotStatePtr particlesVec;
413  for (size_t j = 0; j != numDrawnParticles; ++j) {
414  auto opptBeliefState = particles[j];
415  if (opptBeliefState) {
416  auto userData = opptBeliefState->getUserData();
417  if (userData) {
418  auto previousParticleState = userData->as<RobotStateUserData>()->previousState;
419  auto v = static_cast<const VectorState*>(previousParticleState.get())->asVector();
420  oppt::RobotStateSharedPtr prevS(new oppt::VectorState(v));
421  auto interpolatedState = robotPlanningEnvironment_->getRobot()->getStateSpace()->interpolate(prevS,
422  opptBeliefState,
423  d);
424  particlesVec.push_back(interpolatedState);
425  }
426  }
427  }
428 
429  auto normalizedSubState =
430  robotPlanningEnvironment_->getRobot()->getStateSpace()->normalizeState(subStates[i]);
431 
432  robotExecutionEnvironment_->getRobot()->updateViewer(normalizedSubState,
433  particlesVec,
434  problemEnvironmentOptions_->particleOpacity,
435  false,
436  false);
437  }
438 
439  VectorRobotStatePtr particlesVec(numDrawnParticles);
440  for (size_t i = 0; i != numDrawnParticles; ++i) {
441  if (i < particles.size())
442  particlesVec[i] = particles[i];
443  }

445 #ifdef USE_RVIZ
446  auto environmentInfo = robotExecutionEnvironment_->getEnvironmentInfo();
447  static_cast<ViewerPublisher *>(robotExecutionEnvironment_->getRobot()->getViewer())->updateFromEnvironmentInfo(environmentInfo, true);

451  particlesVec,
452  problemEnvironmentOptions_->particleOpacity,
453  false,
454  true);
455  }
