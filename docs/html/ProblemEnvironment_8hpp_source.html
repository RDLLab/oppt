397  void updateViewer(const RobotStateSharedPtr & currentState,
398  const VectorRobotStatePtr& particles = VectorRobotStatePtr()) {
399  if (!currentState) {
400  LOGGING("currentState is null. Cannot visualize current simulation state");
401  return;
402  }
403  VectorRobotStatePtr subStates = currentState->getSubStates();
404  unsigned int numDrawnParticles = problemEnvironmentOptions_->particlePlotLimit;
405  if (particles.size() < numDrawnParticles)
406  numDrawnParticles = particles.size();
407  for (size_t i = 0; i != subStates.size(); ++i) {
408  FloatType d = ((1.0 / (FloatType)(subStates.size())) * i);
409  VectorRobotStatePtr particlesVec;
410  for (size_t j = 0; j != numDrawnParticles; ++j) {
411  auto opptBeliefState = particles[j];
412  if (opptBeliefState) {
413  auto userData = opptBeliefState->getUserData();
414  if (userData) {
415  auto previousParticleState = userData->as<RobotStateUserData>()->previousState;
416  auto v = static_cast<const VectorState*>(previousParticleState.get())->asVector();
417  oppt::RobotStateSharedPtr prevS(new oppt::VectorState(v));
418  auto interpolatedState = robotPlanningEnvironment_->getRobot()->getStateSpace()->interpolate(prevS,
419  opptBeliefState,
420  d);
421  particlesVec.push_back(interpolatedState);
422  }
423  }
424  }
425 
426  auto normalizedSubState =
427  robotPlanningEnvironment_->getRobot()->getStateSpace()->normalizeState(subStates[i]);
428 
429  robotExecutionEnvironment_->getRobot()->updateViewer(normalizedSubState,
430  particlesVec,
431  problemEnvironmentOptions_->particleOpacity,
432  false,
433  false);
434  }
435 
436  VectorRobotStatePtr particlesVec(numDrawnParticles);
437  for (size_t i = 0; i != numDrawnParticles; ++i) {
438  if (i < particles.size())
439  particlesVec[i] = particles[i];
440  }
441 
442 #ifdef USE_RVIZ
443  auto environmentInfo = robotExecutionEnvironment_->getEnvironmentInfo();
444  static_cast<ViewerPublisher *>(robotExecutionEnvironment_->getRobot()->getViewer())->updateFromEnvironmentInfo(environmentInfo, true);
445 #endif

448  particlesVec,
449  problemEnvironmentOptions_->particleOpacity,
450  false,
451  true);
452  }

766  VectorRobotStatePtr beliefParticles;

772 
773  if (runStartingFn_)
774  runStartingFn_(run);
775 
776  FloatType t0 = oppt::clock_ms();

778  cout << "\nt = " << currentStep << endl;
779  stepResult = StepResult(currentStep, robotExecutionEnvironment_.get());
780  stepResult.currentState = currentState;
781  stepResult.discountFactor = currentDiscount;
782  if (currentStep >= problemEnvironmentOptions_->nSimulationSteps) {
783  // We're out of steps
784  PRINT("Out of steps. Ending simulation run");
785  finalState = currentState;
786  break;
787  }
788

797  LOGGING("Couldn't improve policy. Ending simulation run");
798  break;
799  }
800 
801  totalPlanningTime += (oppt::clock_ms() - planningTimer) / 1000.0;
802 
803  // Get nextAction
804  ActionSharedPtr action = solver_->getNextAction();
805  if (!action) {
806  // No action to execute
807  finalState = currentState;
808  LOGGING("No action for the next step. Ending simulation run");
809  break;

812  cout << "action: " <<

814  << endl;
815  stepResult.action = action;
816

819  propagationRequest->currentState = currentState;
820  propagationRequest->action = action.get();

822

824  robotExecutionEnvironment_->getRobot()->propagateState(propagationRequest);

828  << endl;
829  } else {
830  cout << "Next state: NULL" << endl;
831  }
832 
833  // Get an observation
834  ObservationRequestSharedPtr observationRequest(new ObservationRequest());
835  observationRequest->currentState = propagationResult->nextState;
836  observationRequest->action = action.get();

844  endl;

848 
849  totalDiscountedReward += currentDiscount * reward;
850  stepResult.reward = reward;
851  cout << "Immediate reward: " << reward << endl;
852 
853  // Update the true state
854  currentState = propagationResult->nextState;
855  currentDiscount *= problemEnvironmentOptions_->discountFactor;
856 
857  // Check if we're terminal
858  bool terminal = robotExecutionEnvironment_->isTerminal(propagationResult);
859

861  PRINT("Terminal state reached!");

864  success = true;
865  }

869  // Couldn't update the belief
