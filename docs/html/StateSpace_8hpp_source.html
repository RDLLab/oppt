367  virtual RobotStateSharedPtr normalizeState(const RobotStateSharedPtr& state) const override {
368  VectorFloat lowerStateLimits;
369  VectorFloat upperStateLimits;
370  stateLimits_->getRawLimits()->as<VectorLimitsContainer>()->get(lowerStateLimits, upperStateLimits);
371  VectorFloat stateVec = state->as<VectorState>()->asVector();
372  for (size_t i = 0; i < stateVec.size(); i++) {
373  stateVec[i] = (stateVec[i] - lowerStateLimits[i]) / (upperStateLimits[i] - lowerStateLimits[i]);
374  }
375 
376  RobotStateSharedPtr normalizedState = std::make_shared<VectorState>(stateVec);
377  GazeboWorldStatePtr ws = state->getGazeboWorldState();
378  auto subStates = state->getSubStates();
379  normalizedState->setGazeboWorldState(ws);
380  normalizedState->setSubStates(subStates);
381  return normalizedState;
382  }

384  virtual RobotStateSharedPtr denormalizeState(const RobotStateSharedPtr& state) const override {
385  VectorFloat lowerStateLimits;
386  VectorFloat upperStateLimits;
387  stateLimits_->getRawLimits()->as<VectorLimitsContainer>()->get(lowerStateLimits, upperStateLimits);
388  VectorFloat stateVec = state->as<VectorState>()->asVector();
389  for (size_t i = 0; i < stateVec.size(); i++) {
